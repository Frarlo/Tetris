257  private Quadrato trovaPerno() {
258  Quadrato perno = null;
259  double distanza = 0;
260  for(Quadrato q : quadrati) {
261  double nuovaDistanza = 0;
262  for(Quadrato q2 : quadrati)
263  if(q != q2)
264  nuovaDistanza += q.distanzaDa(q2);
265 
266  if(perno == null || nuovaDistanza < distanza) {
267  perno = q;
268  distanza = nuovaDistanza;
269  } else if(nuovaDistanza == distanza) {
270  perno = q.getPosY() < perno.getPosY() ? q :
271  q.getPosY() > perno.getPosY() ? perno :
272  q.getPosX() < perno.getPosX() ? q : perno;
273  }
274  }
275  return perno;
276  }
